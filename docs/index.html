$('.container .drag')​.draggable({
        containment: ​"parent",
        multiple: ​true,
        selected: ​'.selected',
        beforeStart: ​function () {
            var $this = ​$(this);
            if (!$this​.hasClass('selected')) ​{
                $this​.siblings('.selected')
                    .removeClass('selected');
                $this​.addClass('selected');
            }
        }
    });

    .container {
        position: relative;
    }
    .drag {
        position: absolute;
    }
